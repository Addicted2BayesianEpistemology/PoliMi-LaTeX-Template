% --- 4. Two columns (symmetric) --------------------------------------------
\begin{frame}{Due colonne simmetriche}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Colonna sinistra}\\[0.5em]
      Il template di partenza era un file \texttt{.pptx} di 31 slide,
      formato 16:9, in italiano. Tutte le coordinate sono state estratte
      direttamente dal layout XML.
    \column{0.48\textwidth}
      \textbf{Colonna destra}\\[0.5em]
      I colori, i font e le posizioni sono fedeli all'originale al
      millimetro, così le slide LaTeX hanno lo stesso aspetto del
      file PowerPoint.
  \end{columns}
\end{frame}
